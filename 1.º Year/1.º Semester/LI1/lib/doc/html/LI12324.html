gravidade :: Velocidade #

Vetor velocidade da recColec' :: Jogo -- ^ Jogo -> Jogo -- ^ Jogo com os colecionáveis atualizados recColec' jogo(Jogo m(Mapa m1 m2 m3) i colec((c,pov):r) p(Personagem _ _ pos _ _ _ _ _ moeda dano)) | interColec pos pov && c == Martelo = (Jogo m i r p { aplicaDano = (True,10) } ) | interColec pos pov && c == Moeda = (Jogo m i r p { pontos = (moeda + 69)} ) | interColec pos pov && c == Estrela = jogo | otherwise = (Jogo m i r p) gravidade.

gravidade == (0, 10)
